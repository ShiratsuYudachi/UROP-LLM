N=int(input())
alpha='ABCDEFGHIJKLMNOPQRSTUVWXYZ'
for i in range(0,N):
    a=int(input())
    b=str(input())
    ans=0
    for j in range(0,26):
        if j+1<=b.count(alpha[j]):
            while j+1<=b.count(alpha[j]):
                k=b.find(alpha[j])
                if k<=a-1:
                    b=b[0:k]+'.'+b[k+1:]
                    ans=ans+1
                else: break
        else: break

    print(b,ans)
