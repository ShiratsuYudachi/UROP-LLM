N=int(input())
alpha='ABCDEFGHIJKLMNOPQRSTUVWXYZ'
for i in range(0,N):
    a=int(input())
    b=str(input())
    if len(b)<a:
        for j in range(a-len(b)):
            b+='_'
    else:
        b=b[0:a]
    ans=0
    for j in range(0,26):
        if j+1<=b.count(alpha[j]):
            ans+=1
    print(b,ans)
