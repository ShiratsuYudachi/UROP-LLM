N=int(input())
alpha='ABCDEFGHIJKLMNOPQRSTUVWXYZ'
for i in range(0,N):
    a=int(input())
    b=str(input())
    ans=0
    if a==0:
        print(0)
    else:
        for j in range(0,26):
            if j==0:
                if b.count(alpha[j])>=1:
                    ans+=1
            else:
                if b.count(alpha[j])>=1 and b.count(alpha[j-1])>=1:
                    ans+=1
    print(b,ans)
