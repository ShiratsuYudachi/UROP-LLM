N=int(input())
alpha='ABCDEFGHIJKLMNOPQRSTUVWXYZ'
for i in range(0,N):
    a=int(input())
    b=str(input())
    ans=0
    if len(b)>=1:
        j=0
        while j<=len(b)-1:
            if b[j]==alpha[j]:
                if j+1<=b.count(alpha[j]):
                    ans+=1
            j+=1
    print(b,ans)
