t=int(input())
if t<=10**4:
    for i in range(t):
        n=int(input())
        if n in range(1,2*10**5+1):
            a=[int(ele) for ele in input().split()]
            b=[int(ele) for ele in input().split()]
            d=[[a[i]+b[i],i] for i in range(n)]
            d.sort(key=lambda x:-x[0])
            res=0
            for i in range(n):
                if i%2:res-=(b[d[i][1]]-1)
                else: res+=(b[d[i][1]]-1)
            print(res)
        else:
            print("Value of n not in range")
else:
    print("Value of t not in range")
