t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    c=[]
    for i in range(n):c.append([b[i]-a[i],i])
    if c[0][0]==0:
        pass
    else:
        for x in range(1,n):
            if c[x][0]!=0:break
            else:
                c[x][0]=1e9
    d=c.copy()
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    print(res)
