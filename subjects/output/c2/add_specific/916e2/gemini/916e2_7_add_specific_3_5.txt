t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    d=[[a[i]+b[i],i] for i in range(n)]
    while len(d)>1:
        d.sort()
        if d[0][0]==d[1][0]:d.pop(0)
        else: break

    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    print(res)
