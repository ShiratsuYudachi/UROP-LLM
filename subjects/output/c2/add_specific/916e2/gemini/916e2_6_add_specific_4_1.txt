t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    c=[]
    for i in range(n):c.append([a[i]-b[i],i])
    d=sorted(c,key=lambda x:-x[0])
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    print(res)
