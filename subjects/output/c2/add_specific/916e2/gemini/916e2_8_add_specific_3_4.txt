t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    d=[[a[i]+b[i],i] for i in range(n)]
    d.sort(key=lambda x:-x[0])
    while d:
        if d[0][0]<=0:d.pop(0)
        else:break
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
        for j in range(i+1,n):
            d[j][0]-=b[d[i][1]]
    print(res)
