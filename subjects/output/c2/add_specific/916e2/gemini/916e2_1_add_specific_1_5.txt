def solve():
    t=int(input())
    if t<1 or t>10000:return "Wrong test-case"
    for i in range(t):
        n=int(input())
        if n<2 or n>200000:return "Wrong number of colors"
        a=[int(ele) for ele in input().split()]
        b=[int(ele) for ele in input().split()]
        if not all(1<=x<=1000000000 for x in a):return "Wrong value in Alice's marbles"
        if not all(1<=x<=1000000000 for x in b):return "Wrong value in Bob's marbles"
        d=[[a[i]+b[i],i] for i in range(n)]
        d.sort(key=lambda x:-x[0])
        res=0
        for i in range(n):
            if i%2:res-=(b[d[i][1]]-1)
            else: res+=(b[d[i][1]]-1)
        print(res)
    return "All test cases passed"

else:
    print(solve())
