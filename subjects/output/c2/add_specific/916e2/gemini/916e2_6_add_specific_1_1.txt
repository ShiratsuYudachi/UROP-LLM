t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    if sum(a) == sum(b):
        res = 0
        for i in range(n):b[i]=0
    else:
        d = []
        for j in range(n):
            d.append([a[j]-b[j],j])
        res = (sum(a) - sum(b))//2
        
    d.sort(key=lambda x:-x[0])
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    print(res)
