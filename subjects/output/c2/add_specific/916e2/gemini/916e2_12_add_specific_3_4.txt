t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    d=[[a[i]+b[i],i] for i in range(n)]
    d.sort(key=lambda x:-x[0])
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    while d:
        if a[d[-1][1]]>0:
            if i%2:res-=1
            else: res+=1
            a[d[-1][1]]-=1
        del d[-1]
    print(res)
