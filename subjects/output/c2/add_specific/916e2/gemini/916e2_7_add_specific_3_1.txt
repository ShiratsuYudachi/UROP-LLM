t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    d=[[a[i]+b[i],i] for i in range(n)]
    while d:
        d.sort(reverse=True)
        if d[0][0]==0:break
        res=d.pop(0)[1]
        for j in range(len(d)):
            d[j][0]-=b[res]
            if d[j][0]==0:break
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    print(res)
