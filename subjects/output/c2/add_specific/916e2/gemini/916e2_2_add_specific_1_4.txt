t=int(input())
if t<=10000 and 2<=n<=200000 and 1<=a[i]<=1000000000 and 1<=b[i]<=1000000000:
    pass
else:
    exit()

for _ in range(t):

    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    d=[[a[i]+b[i],i] for i in range(n)]
    d.sort(key=lambda x:-x[0])
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else: res+=(b[d[i][1]]-1)
    print(res)
