t = int(input())
if t <= 0 or t > 1000:
    print("Number of test cases must be between 1 and 1000")
else:
    for _ in range(t):
        n, k = map(int, input().split())
        if not (2 <= n <= 50 and 0 <= k <= n - 1):
            print("Invalid input. n must be between 2 and 50, and k must be between 0 and n-1")
            continue

        s = ''
        if k == 0:
            for i in range(n):
                s = s + ' ' + str(n - i)
            print(s.strip())
            continue
        if k == 1:  # bug
            for i in range(1, n + 1):
                s = s + ' ' + str(i)
            print(s.strip())
            continue
        i = 1
        for j in range(k + 1):
            s = s + ' ' + str(i)
            i = i + 1
        s = s.strip()
        while i <= n:
            s = str(i) + ' ' + s
            i = i + 1
        print(s.strip())
