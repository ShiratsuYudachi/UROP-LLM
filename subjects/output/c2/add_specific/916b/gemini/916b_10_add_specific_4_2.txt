t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        print(s.strip())
        continue
    i=1;
    lst = list(range(1,n+1))
    if k==n-1:
        for i in range(1,n+1):
            s = s +' '+str(i)
        print(s.strip())
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        lst.pop(0)
        i = i+1
    s = s.strip()
    while lst:
        a = lst.pop()
        s = str(a)+' '+s

    print(s.strip())
