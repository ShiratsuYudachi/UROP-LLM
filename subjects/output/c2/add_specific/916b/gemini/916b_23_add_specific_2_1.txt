t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        if s.count('1')>1:
            s = s.replace('1',' 1',1)
        continue
    if k==1:  #bug
        for i in range(1,n+1):
            s = s +' '+str(i)
        if s.count(str(n))>1:
            s= s.replace(str(n),' '+str(n),1)
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        i = i+1
    s = s.strip()
    while i<=n:
        s = str(i)+' '+s
        i = i+1
    if s.count(str(n-k))>1:
        s = s.replace(str(n-k),' '+str(n-k),1)
    if s.count(str(n))>1:
        s = s.replace(str(n),' '+str(n),1)
    print(s)
