t = int(input())
for _ in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        s = s.ljust(n,' '+str(k))
        print(s)
        continue
    if k==1:  #bug
        for i in range(1,n+1):
            s = s +' '+str(i)
        s = s.rjust(n,' '+str(k))
        print(s)
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        i = i+1
    s = s.ljust(n,' '+str(k))
    while i<=n:
        s = str(i)+' '+s
        i = i+1
    s = s.ljust(n,' '+str(k))
    print(s)
