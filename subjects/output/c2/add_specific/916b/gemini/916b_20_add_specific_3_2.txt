t = int(input())
for _ in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        print(s.stripped())
        continue
    if k==1:
        for i in range(1,n+1):
            s = s +' '+str(i)
        print(s.stripped())
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        i = i+1

    s = s.stripped()
    while len(s.split(' ')) < n:
        s1 = s.split(' ')[::-1]
        s1='+'.join(s1)
        s = s1+s
        i = i+1
    print(s.stripped())
