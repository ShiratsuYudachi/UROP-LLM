t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    if k>1:
        s = ''
        i=1
        for j in range(k+1):
            s = s +' '+ str(i)
            i = i+1
        s = s.strip()
        while i<=n:
            s = str(i)+' '+s
            i = i+1
        print(s.strip())
        continue
    else:
        s = ''
        i=1
        while i<=n:
            if i%2==0 and i+1<=n:
                s = s +' '+ str(i+1)
                s = s +' '+ str(i)
                i = i+2
            else:
                s = s+' '+str(i)
                i = i+1
        print(s.strip())
