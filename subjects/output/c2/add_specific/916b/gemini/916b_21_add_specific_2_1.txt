t = int(input())
for _ in range(t):
    n, k = map(int, input().split())
    s = ''
    if k == 0:
        for i in range(n):
            s = s + ' ' + str(n - i)
        print(s.ljust(n))
        continue
    if k == 1:
        for i in range(1, n + 1):
            s = s + ' ' + str(i)
        print(s.ljust(n))
        continue
    i = 1
    for j in range(k + 1):
        s = s + ' ' + str(i)
        i = i + 1
    s = s.ljust(n)
    while i <= n:
        if i + 1 < n and i + 1 <= k:
            s = s + ' ' + str(i + 1)
            i = i + 1
            continue
        s = s + ' ' + str(i)
        i = i + 1
    print(s)
