t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    if k!=(n-1):
        s = ''
        i=1
        for j in range(k+1):
            s = s +' '+ str(i)
            i = i+1
        s = s.strip()
        while i<=n:
            s = str(i)+' '+s
            i = i+1
        print(s.strip())
    else:
        s = ''
        i=n
        for j in range(k+1):
            s = s +' '+ str(i)
            i = i-1
        s = s.strip()
        while i>=1:
            s = str(i)+' '+s
            i = i-1
        print(s.strip())
