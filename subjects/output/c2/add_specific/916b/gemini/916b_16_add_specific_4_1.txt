t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        print(s.strip())
        continue
    if k==1:
        for i in range(1,n+1):
            s = s +' '+str(i)
        print(s.strip())
        continue
    i=1
    if k==n-1:
        s = s + str(n) +' '
        for i in range(1,n):
            s = s +' '+ str(i)
        i = i+1
    else:
        for j in range(1,k+1):
            s = s +' '+ str(i)
            i = i+1
        s = s +' '+ str(min(i+k,n)) +' '
        for j in range(min(i+k,n),n+1):
            s = s +' '+ str(j)
        print(s.strip())
