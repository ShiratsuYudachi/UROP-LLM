t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        while s.count('1')>1:
            i = s.index('1 1')
            s = s[:i]+'1 '+s[i+2:]
        print(s.strip())
        continue
    if k==1:  #bug
        for i in range(1,n+1):
            s = s +' '+str(i)
        while s.count('1 2')>1:
            i = s.index('1 2')
            s = s[:i]+'2 1'+s[i+2:]
        print(s.strip())
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        i = i+1
    s = s.strip()
    while i<=n:
        s = str(i)+' '+s
        i = i+1
    while s.count('1 2')>1:
        i = s.index('1 2')
        s = s[:i]+'2 1'+s[i+2:]
    print(s.strip())
