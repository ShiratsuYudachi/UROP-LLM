t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        while len(s.split())>2:
            print(s.split()[len(s.split())-1]+' '+s.split()[1]+s[2:])
            s=s.split()[0]+' '+s.split()[-1]
        continue
    if k==1:  #bug
        for i in range(1,n+1):
            s = s +' '+str(i)
        while len(s.split())>2:
            print(s.split()[len(s.split())-1]+' '+s.split()[1]+s[2:])
            s=s.split()[0]+' '+s.split()[-1]
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        i = i+1
    s = s.strip()
    while i<=n:
        s = str(i)+' '+s
        i = i+1
    print(s.strip())
