t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        if k==n-1:
            s = s [2:-1]
        pass
    else:
        if k==1:
            for i in range(1,n+1):
                s = s +' '+str(i)
            if k==n-1:
                s = s [2:-1]
            pass
        else:

            continue
        i=1
        for j in range(k+1):
            s = s +' '+ str(i)
            i = i+1
        s = s.strip()
        while i<=n:
            s = str(i)+' '+s
            i = i+1
        if k==n-1:
            s = s [2:-1]
        else:
            s = s [1:-1]

    print(s)
