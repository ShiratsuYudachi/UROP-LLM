t = int(input())
for nt in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        s = s.strip()
        print(s)

    elif k==1:
        for i in range(1,n+1):
            s = s +' '+str(i)
        s = s.strip()
        print(s)

    elif k==2:
        for i in range(1,n+1):
            s = s +' '+str(i)
        for i in range(n-2,-1,-1):
            s = str(s[i+1])+' '+s[:i+1]
        s = s.strip()
        print(s)

    else:
        for i in range(1,n+1):
            s = s +' '+str(i)
        s = s.strip()
        for i in range(n-k,-1,-1):
            s = str(s[i+1])+' '+s[:i+1]
        s = s.strip()
        print(s)
