t = int(input())
if t>=1 and t<=1000:
    for _ in range(t):
        n,k = map(int,input().split())
        if n>=2 and n<=50 and k>=0 and k<=n-1:
            s=''
            if k==0:
                for i in range(n):
                    s = s +' '+str(n-i)
                print(s.strip())
                continue
            if k==1:  #bug
                for i in range(1,n+1):
                    s = s +' '+str(i)
                print(s.strip())
                continue
            i=1
            for j in range(k+1):
                s = s +' '+ str(i)
                i = i+1
            s = s.strip()
            while i<=n:
                s = str(i)+' '+s
                i = i+1
            print(s.strip())
