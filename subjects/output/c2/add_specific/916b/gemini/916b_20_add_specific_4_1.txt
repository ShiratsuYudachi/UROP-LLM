t = int(input())
for _ in range(t):
    n,k = map(int,input().split())
    s=''
    if k==0:
        for i in range(n):
            s = s +' '+str(n-i)
        print(s.ltrim())
        continue
    if k==1:
        for i in range(1,n+1):
            s = s +' '+str(i)
        print(s.ltrim())
        continue
    i=1
    for j in range(k+1):
        s = s +' '+ str(i)
        i = i+1
    s = s.ltrim()
    for j in range(n-i):
        s = s +' '+str(i+j)
    s = s +' '+str(i)
    print(s.ltrim())
