for _ in range(int(input())):
    n, k = map(int, input().split())
    if n <= k:
        a = list(map(int, input().split()))
        b = list(map(int, input().split()))
        a.sort(reverse=True)
        b.sort(reverse=True)
        result = sum(a[:k]) + sum(b[:k - 1])
    else:
        a = list(map(int, input().split()))
        b = list(map(int, input().split()))
        a.sort()
        b.sort()
        max = 0
        sum = 0
        for i in range(min([n - k, len(b)])):
            if i < n - k:
                sum += a[i]
            max = max(list(max + b) + b)
            result = max(result, sum + max * (k - i - 1))
    print(result)
