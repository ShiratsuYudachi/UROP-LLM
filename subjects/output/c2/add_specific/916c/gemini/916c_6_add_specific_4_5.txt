for _ in range(int(input())):
    n, k = map(int, input().split())
    a = list(map(int, input().split()))
    b = list(map(int, input().split()))
    result = 0
    completed = []  # Keep track of completed quests
    for i in range(min([k, n])):
        if i < n:
            sum += a[i]
        maxB = max(list(b[0:i + 1]))  # Maximum value of b among completed quests
        result = max(result, sum + maxB * (k - i - 1))  # Update result with the maximum possible experience
        completed.append(i)
    print(result)
