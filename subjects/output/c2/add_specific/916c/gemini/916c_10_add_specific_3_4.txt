for _ in range(int(input())):
    n, k = map(int, input().split())
    a = list(map(int, input().split()))
    b = list(map(int, input().split()))
    result = 0
    maxB = 0
    sum = 0
    for i in range(min([k, n])):
        if i < n:
            while i > 0 and b[i] < b[i - 1]:
                b[i], b[i - 1] = b[i - 1], b[i]
                a[i], a[i - 1] = a[i - 1], a[i]
                i -= 1

        maxB = max(list(b[0:i + 1]))
        result = max(result, sum + maxB * (k - i - 1))
        sum += a[i]
    print(result)
