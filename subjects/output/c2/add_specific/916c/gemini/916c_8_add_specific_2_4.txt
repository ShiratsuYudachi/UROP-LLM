for _ in range(int(input())):
    n, k = map(int, input().split())
    a = list(map(int, input().split()))
    b = list(map(int, input().split()))
    result = 0
    maxB = 0
    sum = 0
    if k >= n:
        sum = list(a)
        sum.sort(reverse=True)
        result = sum[0:k]
    else:
        for i in range(n):
            if i < k:
                sum += a[i]
        maxB = max(list(b[0:k + 1])) 
        result = max(result, sum + maxB * (k - i - 1))
    print(result)
