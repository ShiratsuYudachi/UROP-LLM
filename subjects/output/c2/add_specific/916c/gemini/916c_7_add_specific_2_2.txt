for _ in range(int(input())):
    n, k = map(int, input().split())
    a = list(map(int, input().split()))
    b = list(map(int, input().split()))
    result = 0
    maxB = 0
    if k >= n:
        sum = 0
        for i in range(n):
            sum += a[i]
        print(sum)
    else:
        sum = 0
        if k < n:
            for i in range(k):
                sum += a[i]
        maxB = max(list(b[0:k]))
        result = max(result, sum + maxB * (k - 1))
    print(result)
